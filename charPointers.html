#include <stdio.h>

void charArray(char *a, int size)// char a[] = "abc";

{

for(int i = 0; i < size; i++)

{

printf("%d: %c \n", i, a[i]);

}

a[0] = 'x';

}

void doubleArray(double dd[], int size)

{

for(int i = 0; i < size; i++)

{

printf("%d: %lf \n", i, dd[i]);

}

dd[0] = 5;

}

int main(int argc, char *argv[]) {

char s[10] = "abc";

char *p = "abc";

double d[3] = {6.7, 5.3, 7.2};

charArray(s, sizeof(s));

printf("%c\n", s[0]);

doubleArray(d, sizeof(d)/ sizeof(double));

printf("%f\n", d[0]);

charArray(p, sizeof(p));

printf("%c\n", p[0]);

return 0;

}
